def has_class_create_action(*args)
  action_module, line = Module.new, __LINE__+1
  action_module.module_eval(%Q{
    def initialize_class_actions(proxy)
      super

      class << proxy
        has_create_action(#{args.map(&:inspect).join(',')})
      end
    end
  }, __FILE__, line)

  extend action_module
end
